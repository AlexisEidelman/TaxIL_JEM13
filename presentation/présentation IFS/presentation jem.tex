\section{Why to be interested in Life-Cycle ?}

\subsection{Inequalities}

\begin{frame}

  \frametitle{Why to be interested in Life-Cycle ?}
  \framesubtitle{\vspace{0.3cm} Evaluation inequalities} \pause
  \begin{itemize}
\item \textbf{Variability of incomes and living standards over life-cycle}
 \begin{enumerate}
\item Analyse canonique du cycle de vie
\item Chocs imprévisibles : situations de chômage, de maladie, d'inactivité, etc. \\ \pause

\end{enumerate}
$\Rightarrow$ Mobilité dans l'échelle des revenus (eg. Slemrod 1992) \\ \pause

  \item \textbf{Vision tronquée des inégalités} \\ \pause
$\Rightarrow$ Distinguer entre inégalités intra- et inter- personnelles
  \end{itemize}
\end{frame}

\begin{frame}

  \frametitle{Les enjeux du cycle de vie}
  \framesubtitle{\vspace{0.3cm}Inégalités sur le cycle de vie : éléments empiriques}
  \begin{itemize}
\item \textbf{Décomposition entre inégalités intra- et inter-individuelles} \pause
 \begin{itemize}
\item Björklund et Palme (1997) : 40 à 50 \% des inégalités de revenus sont intra-personnelles  \pause
 \end{itemize}
\item \textbf{Mesure des inégalités en coupe vs. sur le cycle de vie}
\begin{itemize}
\item Björklund (1993) : diminution du coefficient de Gini de 35 à 40 \%
\item Dépend de la durée considérée
\end{itemize}
  \end{itemize}
\end{frame}

\subsection{Redistribution}

\begin{frame}

  \frametitle{Les enjeux du cycle de vie}
  \framesubtitle{\vspace{0.3cm}Importance pour l'étude de la redistribution} \pause
  \begin{itemize}
\item \textbf{Distinction entre redistribution intra- et inter-personnelle}
	 \begin{itemize}
	\item Redistribution intra-individuelle $\Rightarrow$ reflète une logique d'assurance \pause
	 \end{itemize}
\item \textbf{Deux cas-types extrêmes}
 	\begin{itemize}
	\item Individus aux revenus constants au cours du cycle de vie, différents entre eux\\
 $\Rightarrow$ Redistribution inter-personnelle principalement \pause
	\item Individus parfaitement identiques, aux revenus affectés par des variations au cours du temps \\
 $\Rightarrow$ Pas besoin de redistribution si l'accès au marché du crédit et à  l'assurance est parfait

	\end{itemize}
\end{itemize}
\end{frame}


\begin{frame}

  \frametitle{Les enjeux du cycle de vie}
  \begin{itemize}

 	\begin{itemize}

	\item Impact de la perspective de cycle de vie inégalement réparti dans la distribution des revenus \pause
	\end{itemize}
 \end{itemize}

\subsection{Sources d'information}
\begin{frame}

  \frametitle{I. Sources d'information}
  \begin{itemize}
    \item \textbf{Le problème : } Connaître le passé précisément et avoir autant d'éléments que possible sur le futur.
    \item\textbf{ La réponse de \til \ :}  \pause
    \begin{itemize}
      \item Utilisation de l'enquête patrimoine
      \item Enrichissement par un appariement avec l'EIR-EIC
      \item Fermeture de l'échantillon
      \item Utilisation de différentes études pour les équations de simulations
    \end{itemize}
    \item \textbf{Ce qu'on pourrait faire de mieux : }
    \begin{itemize}
    \item Utiliser d'autres enquêtes comme SIP
    \item Améliorer les simulations (avec de la validation)
    \end{itemize}
  \end{itemize}
\end{frame}

\subsection{Formation des unions}
\begin{frame}

  \frametitle{II. Formation des unions}
  \begin{itemize}
    \item \textbf{Le problème : } unir des individus en respectant les caractéristiques jointes.
    \item \textbf{ La réponse de \til \ :}  \pause
    \begin{itemize}
      \item Distance entre conjoints (pas une préférence)
      \item Algorithme de minimisation
    \end{itemize}
    \item \textbf{Ce qu'on pourrait faire de mieux : }
    \begin{itemize}
    \item Améliorer l'algorithme de minimisation (méthode hongroise)
    \item Faire un tirage aléatoire de conjoints potentiel et sélectionner parmi eux.
    \end{itemize}
  \end{itemize}
\end{frame}



\subsection{Pondérations}
\begin{frame}

  \frametitle{III. Pondérations}
  \begin{itemize}
    \item \textbf{Le problème : } En simulation dynamique, les liens entre individus changent. Comment relier deux individus de pondérations différentes ?
    \item \textbf{ La réponse de \til \ :}  \pause
    \begin{itemize}
      \item Travail avec une pondération uniforme
      \item Duplication des individus selon leur pondération
    \end{itemize}
    \item \textbf{Ce qu'on pourrait faire de mieux : }
    \begin{itemize}
    \item La plus petite pondération de l'enquête patrimoine est 6 ! \\
    $\rightarrow$ 11 millions d'individus
    \item  200 comme pondération uniforme et non 6  \\
    $\rightarrow$ 300 000 individus \\
    \item Augmenter la taille de l'échantillon
    \end{itemize}
  \end{itemize}
\end{frame}

\subsection{Individualisation des transferts}
\begin{frame}

  \frametitle{IV. Individualisation des transferts}
  \begin{itemize}
    \item \textbf{Le problème : } L'unité d'étude ne peut plus être un ménage ou un foyer fiscal mais seulement un individu. Comment individualiser les transferts ?
    \item \textbf{ La réponse de \til \ :}  \pause
    \begin{itemize}
      \item L'impôt sur le revenu est séparé entre déclarant et conjoint...
      \item sauf l'avantage lié au quotient familial qui est réparti entre les personnes à charge.
      \item Les prestations familiales sont réparties entre les enfants
      \item Les autres prestations sont réparties entre les personnes du ménage.
    \end{itemize}
    \item \textbf{Ce qu'on pourrait faire de mieux : }
    \begin{itemize}
    \item Séparer l'IR au prorata des revenus mais...
    \item ...la bonne solution c'est le recours à la valeur de Shapley.
    \end{itemize}
  \end{itemize}
\end{frame}

\subsection{Le pas temporel}
\begin{frame}

  \frametitle{V. Le choix du pas temporel}
  \begin{itemize}
    \item \textbf{Le problème : } Un trade-off entre précision et temps de simulation.
    \item \textbf{ La réponse de \til \ :}  \pause
    \begin{itemize}
      \item Pour l'instant : pas annuel
      \item Possibilité de ne faire tourner certains éléments qu'à certaines périodes
    \end{itemize}
    \item \textbf{Ce qu'on pourrait faire de mieux : }
    \begin{itemize}
    \item Pas mensuel
    \item ...ou au moins trimestriel
    \item automatiser les paramètres en fonction du pas.
    \item Simuler des durées plutôt que des événements ?
    \end{itemize}
  \end{itemize}
\end{frame}

\subsection{Alignement}
\begin{frame}

  \frametitle{VI. Alignement}
  \begin{itemize}
    \item \textbf{Le problème : } Le nombre d'événements (de décès par exemple) prédits par le tirage d'une variable binaire est trop volatil.\\
    Comment maîtriser le nombre d'occurrences ?
    \item \textbf{ La réponse de \til \ :}  \pause
    \begin{itemize}
      \item Utilisation d'une méthode implémentée dans Liam2 (voir Li et O'Donoghue 2012)
    \end{itemize}
    \item \textbf{Ce qu'on pourrait faire de mieux : }
    \begin{itemize}
    \item Tester les différentes méthodes d'alignement.
    \end{itemize}
  \end{itemize}
\end{frame}
